import React, { useState, useEffect } from 'react';
import { Link } from 'react-router-dom';
import { createPageUrl } from './utils';
import { motion, AnimatePresence } from 'framer-motion';
import { FlaskConical, Menu, X, ChevronDown, Beaker, GitCompare, Zap, Activity, BarChart3, Building2, Users, Award, CheckCircle, Settings, FileText } from 'lucide-react';
import { Button } from "@/components/ui/button";
import {
  DropdownMenu,
  DropdownMenuContent,
  DropdownMenuItem,
  DropdownMenuTrigger,
} from "@/components/ui/dropdown-menu";

const navLinks = [
  { name: "Home", href: createPageUrl("Home") },
  { 
    name: "Services", 
    href: "#services",
    submenu: [
      { name: "Phase I Studies", href: createPageUrl("PhaseIClinicalStudies"), icon: Beaker, desc: "Early-phase clinical trials" },
      { name: "Bioequivalence Studies", href: createPageUrl("BioequivalenceStudies"), icon: GitCompare, desc: "Generic drug development" },
      { name: "First-In-Human Trials", href: createPageUrl("FirstInHumanStudies"), icon: Zap, desc: "Novel compound testing" },
      { name: "PK/PD Studies", href: createPageUrl("PKPDStudies"), icon: Activity, desc: "Pharmacokinetics & dynamics" },
      { name: "Bioavailability Studies", href: createPageUrl("BioavailabilityStudies"), icon: BarChart3, desc: "Drug absorption analysis" },
    ]
  },
  { 
    name: "About", 
    href: "#about",
    submenu: [
      { name: "Our Story", href: "#about", icon: Building2, desc: "20 years of excellence" },
      { name: "Team", href: "#about", icon: Users, desc: "Expert investigators" },
      { name: "Why Choose CRS", href: "#about", icon: Award, desc: "Our competitive edge" },
    ]
  },
  { 
    name: "Facilities", 
    href: "#facilities",
    submenu: [
      { name: "Clinical Unit", href: "#facilities", icon: Building2, desc: "70+ bed capacity" },
      { name: "Our Location", href: "#facilities", icon: Award, desc: "Ljubljana, Slovenia" },
      { name: "Equipment", href: "#facilities", icon: Settings, desc: "State-of-the-art tools" },
    ]
  },
  { 
    name: "Capabilities", 
    href: "#capabilities",
    submenu: [
      { name: "Study Management", href: "#capabilities", icon: CheckCircle, desc: "End-to-end oversight" },
      { name: "Quality Assurance", href: "#capabilities", icon: Award, desc: "GCP compliance" },
      { name: "Data Management", href: "#capabilities", icon: FileText, desc: "Statistical analysis" },
    ]
  },
  { name: "Contact", href: "#contact" },
];

export default function Layout({ children }) {
  const [scrolled, setScrolled] = useState(false);
  const [mobileMenuOpen, setMobileMenuOpen] = useState(false);

  useEffect(() => {
    const handleScroll = () => {
      setScrolled(window.scrollY > 50);
    };
    window.addEventListener('scroll', handleScroll);
    return () => window.removeEventListener('scroll', handleScroll);
  }, []);

  const handleNavClick = (href) => {
    setMobileMenuOpen(false);
    if (href.startsWith('#')) {
      const element = document.querySelector(href);
      if (element) {
        element.scrollIntoView({ behavior: 'smooth' });
      }
    } else {
      window.location.href = href;
    }
  };

  return (
    <div className="min-h-screen bg-white">
      {/* Navigation */}
      <header
        className={`fixed top-0 left-0 right-0 z-50 transition-all duration-500 ${
          scrolled 
            ? 'bg-white/95 backdrop-blur-md shadow-sm' 
            : 'bg-transparent'
        }`}
      >
        <nav className="max-w-7xl mx-auto px-6 lg:px-8">
          <div className="flex items-center justify-between h-20">
            {/* Logo */}
            <Link to={createPageUrl('Home')} className="flex items-center gap-3">
              <img 
                src="https://qtrypzzcjebvfcihiynt.supabase.co/storage/v1/object/public/base44-prod/public/69524b12794349e7f2f92e2c/c26816966_CRS--Vizitka-Tanja-Turk--izrez.png" 
                alt="CRS Clinical Research Services" 
                className="h-12 w-auto"
              />
            </Link>

            {/* Desktop Navigation */}
            <div className="hidden lg:flex items-center gap-6 absolute left-1/2 transform -translate-x-1/2">
              {navLinks.map((link) => (
                link.submenu ? (
                  <DropdownMenu key={link.name}>
                    <DropdownMenuTrigger asChild>
                      <button
                        className={`flex items-center gap-1 text-sm font-medium transition-colors duration-300 ${
                          scrolled 
                            ? 'text-slate-600 hover:text-blue-600' 
                            : 'text-slate-300 hover:text-white'
                        }`}
                      >
                        {link.name}
                        <ChevronDown className="h-3 w-3" />
                      </button>
                    </DropdownMenuTrigger>
                    <DropdownMenuContent 
                      align="center" 
                      className="min-w-[600px] bg-white shadow-2xl border border-slate-200 rounded-2xl p-6 mt-2"
                    >
                      <div className="grid grid-cols-1 gap-2">
                        {link.submenu.map((item, idx) => {
                          const Icon = item.icon;
                          return (
                            <DropdownMenuItem 
                              key={item.name}
                              onClick={() => handleNavClick(item.href)}
                              className="cursor-pointer p-0 focus:bg-transparent"
                            >
                              <div className="w-full flex items-start gap-4 p-4 rounded-xl hover:bg-gradient-to-r hover:from-blue-50 hover:to-transparent transition-all duration-300 group">
                                <div className="w-12 h-12 rounded-lg bg-blue-100 group-hover:bg-blue-600 flex items-center justify-center transition-colors duration-300 flex-shrink-0">
                                  <Icon className="h-6 w-6 text-blue-600 group-hover:text-white transition-colors duration-300" />
                                </div>
                                <div className="flex-1">
                                  <div className="font-semibold text-slate-900 group-hover:text-blue-600 transition-colors duration-300 mb-1">
                                    {item.name}
                                  </div>
                                  <div className="text-sm text-slate-500 group-hover:text-slate-600 transition-colors duration-300">
                                    {item.desc}
                                  </div>
                                </div>
                              </div>
                            </DropdownMenuItem>
                          );
                        })}
                      </div>
                    </DropdownMenuContent>
                  </DropdownMenu>
                ) : (
                  <button
                    key={link.name}
                    onClick={() => handleNavClick(link.href)}
                    className={`text-sm font-medium transition-colors duration-300 ${
                      scrolled 
                        ? 'text-slate-600 hover:text-blue-600' 
                        : 'text-slate-300 hover:text-white'
                    }`}
                  >
                    {link.name}
                  </button>
                )
              ))}
            </div>

            {/* Action Buttons */}
            <div className="hidden lg:flex items-center gap-3">
              <Button 
                onClick={() => handleNavClick(createPageUrl("ForStudyParticipants"))}
                className="bg-white hover:bg-slate-100 text-blue-900 rounded-xl px-6 transition-all duration-300"
              >
                Study Participants
              </Button>
              <Button 
                onClick={() => handleNavClick('#contact')}
                className="bg-blue-900 hover:bg-blue-800 text-white rounded-xl px-6 transition-all duration-300"
              >
                Contact Us
              </Button>
            </div>

            {/* Mobile Menu Button */}
            <button
              onClick={() => setMobileMenuOpen(!mobileMenuOpen)}
              className={`lg:hidden p-2 rounded-xl transition-colors ${
                scrolled ? 'text-slate-900' : 'text-white'
              }`}
            >
              {mobileMenuOpen ? <X className="h-6 w-6" /> : <Menu className="h-6 w-6" />}
            </button>
          </div>
        </nav>

        {/* Mobile Menu */}
        <AnimatePresence>
          {mobileMenuOpen && (
            <motion.div
              initial={{ opacity: 0, height: 0 }}
              animate={{ opacity: 1, height: 'auto' }}
              exit={{ opacity: 0, height: 0 }}
              className="lg:hidden bg-white border-t"
            >
              <div className="max-w-7xl mx-auto px-6 py-6 space-y-3">
                {navLinks.map((link) => (
                  <div key={link.name}>
                    <button
                      onClick={() => handleNavClick(link.href)}
                      className="block w-full text-left text-slate-600 hover:text-blue-600 py-2 font-medium"
                    >
                      {link.name}
                    </button>
                    {link.submenu && (
                      <div className="ml-4 mt-2 space-y-2">
                        {link.submenu.map((item) => (
                          <button
                            key={item.name}
                            onClick={() => handleNavClick(item.href)}
                            className="block w-full text-left text-slate-500 hover:text-blue-600 py-1 text-sm"
                          >
                            â€¢ {item.name}
                          </button>
                        ))}
                        </div>
                        )}
                        </div>
                        ))}
                        <Button 
                        onClick={() => handleNavClick(createPageUrl("ForStudyParticipants"))}
                        className="w-full bg-white hover:bg-slate-100 text-blue-900 rounded-xl mt-4 border border-slate-200"
                        >
                        Study Participants
                        </Button>
                        <Button 
                        onClick={() => handleNavClick('#contact')}
                        className="w-full bg-blue-900 hover:bg-blue-800 text-white rounded-xl"
                        >
                        Contact Us
                        </Button>
              </div>
            </motion.div>
          )}
        </AnimatePresence>
      </header>

      {/* Main Content */}
      <main>
        {children}
      </main>
    </div>
  );
}
